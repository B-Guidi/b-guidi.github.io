import json
import os
import numpy as np

import psycopg2

import src.model.util.dimension_handler as d_handl
from src.model.factors import init_factors_matrix, get_level
from src.persistence.database_connection import DatabaseConnection


class PostgreSQLConnection(DatabaseConnection):
    """DatabaseConnection implementation for PostgreSQL database.
    """
    def __init__(self, is_local):
        """Constructor.

        Args:
            is_local: a bool that determines wheter the database runs locally or remotelly.
        """
        if is_local:
            self.conn = psycopg2.connect("dbname=profile_gen_db user=postgres")
        else:
            self.conn = psycopg2.connect(os.environ['DATABASE_URL'], sslmode='require')

    def insert_instances(self, instances, levels_calculated):
        """Inserts the given instances in the database.

        Args:
            instances: a list(Instance) that contains the instances to insert in the database.
            levels_calculated: a bool that determines wheter or not the levels for features where previously calculated.
        Returns:
            None
        """
        with self.conn.cursor() as cursor:
            query = 'INSERT INTO instance(features, dimensions) VALUES (%s, %s)'
            [cursor.execute(query, i.to_json(levels_calculated)) for i in instances]
            self.conn.commit()

    def query_instances(self):
        """Queries the existing instances in the database.

        Returns:
            a dict(str, list(int)), where k is the name of a factor or dimension in the model, and v is a list of values
                for it.
        """
        cursor = self.conn.cursor()
        cursor.execute('SELECT * FROM instance')

        feat_dict = init_factors_matrix()  # The cols of the matrix are the dict keys (f names). The rows are instances.
        dim_dict = d_handl.copy_all_names()

        for i in cursor.fetchall():
            for f_name, f_vals in feat_dict.items():
                level = np.NaN if i[1][f_name] == 'NaN' else get_level(f_name, i[1][f_name])  # i[1] contains features.
                f_vals.append(level)
            [d_vals.append(i[2][d_name]) for d_name, d_vals in dim_dict.items()]  # i[2] contains dimensions.

        return {**feat_dict, **dim_dict}

    def insert_events(self, events):
        """Inserts the given events in the database.

        Args:
            events: a list({'user_id': int, 'factor_id': int, 'value': Any}) with events to insert in the database.
        Returns:
            None
        """
        formated_events = []
        [formated_events.append(json.dumps(e)) for e in events]

        with self.conn.cursor() as cursor:
            query = 'INSERT INTO event(event_info) VALUES (%s)'
            for e in formated_events:  # TODO Try to make the inserts all at once.
                cursor.execute(query, (e, ))
            self.conn.commit()

        return len(events)

    def query_events(self):
        """Queries the existing events in the database.

        Returns:
            a list({'user_id': int, 'factor_id': int, 'value': Any}) that contains all the events stored in the
                database.
        """
        cursor = self.conn.cursor()
        cursor.execute('SELECT event_info FROM event')

        query_result = []
        [query_result.append(e[0]) for e in cursor.fetchall()]

        return query_result

    def delete_existing_events(self):
        """Deletes all the existing events in the database.

        Returns:
            None
        """
        cursor = self.conn.cursor()
        cursor.execute('DELETE FROM event')
        self.conn.commit()
