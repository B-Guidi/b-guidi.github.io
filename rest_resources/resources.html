import json

from flask import request
from flask_restful import Resource

from src.model import mediator


class Events(Resource):
    """
    Resource for handling events.
    """
    def put(self):
        """
        Receives events and persist them in the database.

        JSON format required:
            {0: {'user_id': 0, 'factor_id': 0, 'value': 88},
            1: {'user_id': 0, 'factor_id': 0, 'value': 79},
            2: {'user_id': 0, 'factor_id': 1, 'value': 145},
            3: {'user_id': 0, 'factor_id': 1, 'value': 75},
            4: {'user_id': 0, 'factor_id': 2, 'value': 5},
            5: {'user_id': 0, 'factor_id': 3, 'value': 'Audio'},
            6: {'user_id': 0, 'factor_id': 4, 'value': 130},
            7: {'user_id': 0, 'factor_id': 4, 'value': 110}}
        """
        events = []
        [events.append(e) for e in json.loads(request.json).values()]

        return json.dumps(mediator.insert_events(events)), 200


class Instances(Resource):
    """
    Resource for handling instances.
    """
    def put(self):
        """
        Receives instances, persist them in the database and then retrains the model.

        JSON format required:
            {0: {'features': {'TiempoLecturaUserStory': 12, 'TiempoTrabajoUserStory': 31, 'ParticipacionesMeetings': 6,
                     'Recurso': 'Texto', 'LineasCodigo': 45},
                'dimensions': {'Percepcion': -2, 'Procesamiento': -1, 'XYZ': 0, 'Entendimiento': 3}},
            1: {'features': {'TiempoLecturaUserStory': 75, 'TiempoTrabajoUserStory': 125, 'ParticipacionesMeetings': 9,
                             'Recurso': 'Texto', 'LineasCodigo': 73},
                'dimensions': {'Percepcion': 2, 'Procesamiento': 0, 'XYZ': 0, 'Entendimiento': -1}},
            2: {'features': {'TiempoLecturaUserStory': 65, 'TiempoTrabajoUserStory': 146, 'ParticipacionesMeetings': 12,
                             'Recurso': 'Imagen', 'LineasCodigo': 147},
                'dimensions': {'Percepcion': 3, 'Procesamiento': 2, 'XYZ': 1, 'Entendimiento': -3}}}
        """
        instances = []
        [instances.append(i) for i in json.loads(request.json).values()]
        return json.dumps(mediator.insert_instances(instances)), 200


class ModelUpdate(Resource):
    """
    Resource for handling the processing of events by the app.
    """
    def post(self):
        """
        Process the stored event data, generates instances througth them and then retrains the model with them.
        """
        result = mediator.process_events()
        print(result)
        return json.dumps(result[0]), result[1]
