from src.model.inference import clasify_new_instances
from src.model.inference.bayesian_net import BayesianNetwork
from src.model.inference.inference_algorithm import NearestNeighbour
from src.model.instances import process_event_data, process_instances, merge_instances
from src.persistence.postgresql_connection import PostgreSQLConnection
from src.model.factors import get_total_factors, exist_factor, get_factor_name, is_valid_event_value

"""
Minimal number of new instances required to classify with the bayesian network.
"""
inf_threshold = 40

bayesian_net = BayesianNetwork(
    [(0, 0),  # TiempoLecturaUS -> Percepcion.
     (1, 0),  # TiempoTrabajoUS -> Percepcion.
     (4, 0),  # LineasCodigo -> Percepcion.
     (4, 1),  # LineasCodigo -> Entendimiento.
     (2, 2),  # ParticipacionesMeetings -> Procesamiento.
     (3, 3)])  # Recurso -> Recepcion.
aux_inference_mechanism = NearestNeighbour()

"""
Class that handles data persistence.
"""
db_conn = PostgreSQLConnection(is_local=True)


def __insert_and_fit(new_instances, levels_calculated, existing_instances=None):
    """
    Inserts new instances in the database, merges them with the existing ones, and then retrains the model.

    :param new_instances: list(Instance) that contains the new instances to insert.
    :param levels_calculated: boolean that determines wheter the levels for the features where previously calculated or
        not.
    :param existing_instances: dict(str, list(int)) where k is a f or dimension name, and v is a list with values
        for the given name. If None, then the newinstances must be obtained from the database.
    :return: None
    """
    # Queries existing new instances from the DB, in case they were not previously queried.
    existing_instances = db_conn.query_instances() if existing_instances is None else existing_instances

    # Insert new
    # instances in the DB.
    db_conn.insert_instances(new_instances, levels_calculated=levels_calculated)

    # Merge existing and new instances.
    merge_instances(dest=existing_instances, src=new_instances, src_levels_calculated=levels_calculated)

    # Only fits the BN in case the inferior threshold is surpassed, to avoid innecesary computing.
    if get_total_factors() > 0 and len(existing_instances[get_factor_name(0)]) >= inf_threshold:
        bayesian_net.fit(existing_instances, print_data=False)


def insert_instances(raw_instances_data):
    """
    Inserts new instances in the model and then update it.

    :param raw_instances_data: list(dict(str,dict(str, int))) that contains raw instances data.
    :return: dict that contains the result of the operation, indicating the number of new instances succesfully added.
    """
    size_new_instances = len(raw_instances_data)

    # Formats and validates input. Removes new instances with invalid values.
    raw_instances_data = process_instances(raw_instances_data)

    # Inserts new instances into the db and then retrain the BN.
    __insert_and_fit(raw_instances_data, levels_calculated=False)

    return {'result': "{} out of {} new instances were inserted.".format(len(raw_instances_data), size_new_instances)}


def process_events():
    """
    Queries events from the database, generates new instances through them, and then updates the model.

    :return: dict that contains the result of the operation, indicating wheter there was an error or not.
    """
    # Queries events from db, and transform them into valid new instances for the BN.
    new_instances = process_event_data(db_conn.query_events())

    # Queries existing new instances from the DB.
    existing_instances = db_conn.query_instances()

    # Check that avoids problems in case the model has no factors in it.v
    if get_total_factors() == 0:
        return {'result': 'The instance clasifying could not be made because there is no factors in the model.'}, 400

    # Check that avoids problems in case the model has no new instances in it.
    if len(next(iter(existing_instances.values()))) == 0:
        msg = 'The instance clasifying could not be made because there is no new instances in the model.'
        return {'result': msg}, 400

    # Clasifies the new instances.
    new_instances = clasify_new_instances(existing_instances, bayesian_net, aux_inference_mechanism, inf_threshold,
                                          new_instances)

    # Inserts the new instances and then fit the BN.
    __insert_and_fit(new_instances, levels_calculated=True, existing_instances=existing_instances)

    # Clears the event table, because the existing ones were transformed into new instances.
    db_conn.delete_existing_events()

    msg = "{} new instances were inserted, and the model was retrained.".format(len(new_instances))
    return {'result': msg}, 200


def insert_events(new_events):
    """
    Inserts the given events into the database, to be used when model is updated.

    :param new_events: list({'user_id': int, 'factor_id': int, 'value': Any}) with events to insert in the database.
    """
    valid_events = []
    for e in new_events:
        if exist_factor(e['factor_id']) and is_valid_event_value(e['factor_id'], e['value']):
            valid_events.append(e)

    ins_events = db_conn.insert_events(valid_events)

    return {'result': "{} out of {} events were inserted.".format(ins_events, len(new_events))}
