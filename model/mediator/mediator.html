from src.model.factors import get_total_factors, exist_factor, get_factor_name, is_valid_event_value
from src.model.inference import clasify_new_instances
from src.model.inference.bayesian_net import BayesianNetwork
from src.model.instances import process_event_data, process_instances, merge_instances


class Mediator:
    """
    Class that handles the main functionalities of the system.
    """

    def __init__(self, bn_edges, inf_threshold, aux_inference_algorithm, database_conn):
        """Constructor.

        Args:
            bn_edges: a list(tuple(int, int) that represents the dependencies that exist between factors and dimensions.
            inf_threshold: an int that represents the minimal number of instances required to use the bayesian network
                for classification.
            aux_inference_algorithm: an InferenceAlgorithm object used when the threshold is not surpassed.
            database_conn: a DatabaseConnection object that handles persistence and querying of data.
        """
        self.__inf_threshold = inf_threshold
        self.__bayesian_net = BayesianNetwork(bn_edges)
        self.__aux_inference_algorithm = aux_inference_algorithm
        self.__db_conn = database_conn

    def __insert_and_fit(self, new_instances, levels_calculated, existing_instances=None):
        """Inserts new instances in the database, merges them with the existing ones, and then retrains the model.

        Args:
            new_instances: a list(Instance) that contains the new instances to insert.
            levels_calculated: a bool that determines wheter the levels for the features where previously calculated or
                not.
            existing_instances: a dict(str, list(int)) where k is a factor or dimension name, and v is a list with
                values for the given name. If None, then the newinstances must be obtained from the database.

        Returns:
            None
        """
        # Queries existing new instances from the DB, in case they were not previously queried.
        existing_instances = self.__db_conn.query_instances() if existing_instances is None else existing_instances

        # Insert new
        # instances in the DB.
        self.__db_conn.insert_instances(new_instances, levels_calculated=levels_calculated)

        # Merge existing and new instances.
        merge_instances(dest=existing_instances, src=new_instances, src_levels_calculated=levels_calculated)

        # Only fits the BN in case the inferior threshold is surpassed, to avoid innecesary computing.
        if get_total_factors() > 0 and len(existing_instances[get_factor_name(0)]) >= self.__inf_threshold:
            self.__bayesian_net.fit(existing_instances, print_data=False)

    def insert_instances(self, raw_instances_data):
        """Inserts new instances in the model and then update it.

        Args:
            raw_instances_data: a list(dict(str,dict(str, int))) that contains raw instances data.

        Returns:
            the number of instances added to the model.
        """
        size_new_instances = len(raw_instances_data)

        # Formats and validates input. Removes new instances with invalid values.
        raw_instances_data = process_instances(raw_instances_data)

        # Inserts new instances into the db and then retrain the BN.
        self.__insert_and_fit(raw_instances_data, levels_calculated=False)

        return {'result': "{} out of {} new instances were inserted."
                .format(len(raw_instances_data), size_new_instances)}

    def process_events(self):
        """Queries events from the database, generates new instances through them, and then updates the model.

        Returns:
             the number of instances added to the model.
        """
        # Queries events from db, and transform them into valid new instances for the BN.
        new_instances = process_event_data(self.__db_conn.query_events())

        # Queries existing new instances from the DB.
        existing_instances = self.__db_conn.query_instances()

        # Check that avoids problems in case the model has no factors in it.v
        if get_total_factors() == 0:
            msg = 'The instance clasifying could not be made because there is no factors in the model.'
            return {'result': msg}, 400

        # Check that avoids problems in case the model has no new instances in it.
        if len(next(iter(existing_instances.values()))) == 0:
            msg = 'The instance clasifying could not be made because there is no new instances in the model.'
            return {'result': msg}, 400

        # Clasifies the new instances.
        new_instances = clasify_new_instances(existing_instances, self.__bayesian_net, self.__aux_inference_algorithm,
                                              self.__inf_threshold, new_instances)

        # Inserts the new instances and then fit the BN.
        self.__insert_and_fit(new_instances, levels_calculated=True, existing_instances=existing_instances)

        # Clears the event table, because the existing ones were transformed into new instances.
        self.__db_conn.delete_existing_events()

        msg = "{} new instances were inserted, and the model was retrained.".format(len(new_instances))
        return {'result': msg}, 200

    def insert_events(self, new_events):
        """Inserts the given events into the database, to be used when model is updated.

        Args:
            new_events: a list({'user_id': int, 'factor_id': int, 'value': Any}) with events to insert in the
                database.

        Returns:
            the number of events added to the model.
        """
        valid_events = []
        for e in new_events:
            if exist_factor(e['factor_id']) and is_valid_event_value(e['factor_id'], e['value']):
                valid_events.append(e)

        ins_events = self.__db_conn.insert_events(valid_events)

        return {'result': "{} out of {} events were inserted.".format(ins_events, len(new_events))}
