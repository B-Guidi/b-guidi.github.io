import pandas as pd
from pgmpy.factors.discrete import TabularCPD
from pgmpy.inference import VariableElimination
from pgmpy.models import BayesianModel

import src.model.util.dimension_handler as d_handl
from src.model.factors import get_factor_name, get_cardinality


class BayesianNetwork:
    """Bayesian network builded on top of BayesianModel class provided by pgmpy framework.
    """

    def __init__(self, edges):
        """Initialize a Bayesian Network.

        Args:
            edges: a list(tuple(int, int)), where the first element of the tuple is a factor, and the second one is a
                dimension. Each tuple represents the influence of a factor over a dimension.
        """
        n_edges = []
        [n_edges.append((get_factor_name(f_id), d_handl.get_dimension_name(d_id))) for f_id, d_id in edges]

        self.model = BayesianModel(n_edges)

        # Saves required existing_instances from dimension to create the CPDs.
        d_dict, f_set = {}, set()
        for n_edge, edge in zip(n_edges, edges):
            f_name, d_name, f_id, d_id = n_edge[0], n_edge[1], edge[0], edge[1]

            f_set.update({f_id})
            if d_name not in d_dict:
                d_dict.update({d_name: {'variable': d_name, 'variable_card': d_handl.get_dimension_card(d_id),
                                        'evidence': [], 'evidence_card': []}})
            d_dict[d_name]['evidence'].append(f_name)
            d_dict[d_name]['evidence_card'].append(get_cardinality(f_id))

        # Adds to the model the CPDs associated with the factors.
        for f_id in f_set:
            f_lv = get_cardinality(f_id)
            f_prob = round(1 / f_lv, 3)

            self.model.add_cpds(TabularCPD(
                variable=get_factor_name(f_id),
                variable_card=f_lv,
                values=[[f_prob]] * f_lv))

        # Adds to the model the CPDs associated with the dimensions.
        for k, v in d_dict.items():
            d_prob = round(1 / v['variable_card'], 3)
            cols = 1
            for card in v['evidence_card']:
                cols = cols * card

            self.model.add_cpds(TabularCPD(
                variable=v['variable'],
                variable_card=v['variable_card'],
                values=[[d_prob] * cols] * v['variable_card'],
                evidence=v['evidence'],
                evidence_card=v['evidence_card']))

        if self.model.check_model():
            print("Model correctly initialized")

    def fit(self, existing_instances, print_data=False):
        """Formats the given data into a pandas DataFrame, and then fits the model with it.

        Args:
            existing_instances: dict(str,list(int)), where k is the name of a factor or dimension in the model, and
                v is a list of values for it. If a node is in the model but not in the dict, then the corresponding
                column will be filled with numpy.NaN.
            print_data: bool that indicates if the pandas DataFrame will be printed or not. False by default.
        Returns:
            None
        Raises:
            ValueError if the lists in the given data do not have the same length.
        """
        try:
            existing_instances = pd.concat([pd.DataFrame(data=existing_instances)], ignore_index=True)
            self.model.fit(existing_instances)

            if print_data:
                print(existing_instances)

        except ValueError:
            length_l = []
            [length_l.append(len(v)) for v in existing_instances.values()]
            raise ValueError("All the lists must have the same length ({}).".format(str(length_l)))

    def infer_dimension(self, dimension, named_feats):
        """Infers the dimension taking into account the factors level. The choosen one is the level with maximum
        probability of ocurrence.

        Args:
            dimension: a str that represents the dimension that is going to be infered.
            named_feats: a dict(str, int) that contains features used to infer the dimension, and their levels.
        Returns:
            a tuple(int, float), where the first element is the inferred level dimension, and the second is the
                probability for it.
        """
        model_infer = VariableElimination(self.model)

        # Infers the probabilities for the given dimension.
        dim_probs = model_infer.query(variables=[dimension], evidence=named_feats, show_progress=False).values

        # Gets the dimension lv with maximum probability, and rounds it.
        max_i, max_p = -1, -1
        for i in range(len(dim_probs)):
            if max_p < dim_probs[i]:
                max_i, max_p = i, dim_probs[i]
        return max_i, round(max_p, 3)

    def infer_dimensions(self, dimensions, named_features):
        """Infers the levels for all the given dimensions.

        Args:
            dimensions: a list(str) with the dimensions that are going to be inferred.
            named_features: a dict(str, int) that contains features used to infer the dimension, and their levels.
        Returns:
            a dict(str, int), where k is the name of the dimension, and v is the corresponding level.
        """
        inference = {}
        for d in dimensions:
            inference.update({d: self.infer_dimension(d, named_features)[0]})  # Dim lv at [0]
        return inference
