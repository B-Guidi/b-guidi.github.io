"""
Python module that contains classes and utility methods for infering dimensions and clasifying instances given a set of
features.
"""
from src.model.util.dimension_handler import copy_all_names as copy_all_dims_names


def clasify_new_instances(existing_instances, bayesian_net, aux_inference_algorithm, inf_threshold, new_instances):
    """For each unclasified instance received, infers the values of the dimensions according to their features.

    Args:
        existing_instances: a dict(str, list(int)), where k is the name of a f or dimension in the model, and v is
            a list of values for it.
        bayesian_net: a BayesianNetwork object used to clasify instances.
        aux_inference_algorithm: an InferenceAlgorithm implementation used when the inferior threshold is not reached.
        inf_threshold: an int that contains the minimal number of instances required to clasify with the BN.
        new_instances: a list(Instance) that contains the new instances to insert. The instances are not classified.
    Returns:
        a list(dict(str,tuple(int, int)) + dict(str,int)), that contains all the clasified instances. The key of
            the first dict is a f name, and the value is a tuple with the f level and the f value. The key of
            the second dict is a dimension name, and the value is the infered value for it.
    """
    for i in new_instances:
        # We must infer the dimensions for the given features. f_tuple[0] has the level for the f.
        named_feats = {}
        [named_feats.update({f_name: f_tuple[0]}) for f_name, f_tuple in i.get_features().items()]

        user_dims = __clasify_instance(existing_instances, bayesian_net, aux_inference_algorithm, inf_threshold,
                                       named_feats)

        # Add the infered dimensions for the new instances into the learning set.
        i.set_dimensions(user_dims)

    return new_instances


def __clasify_instance(existing_instances, bayesian_net, aux_inference_algorithm, inf_threshold, named_feats):
    """Infers the dimensions values according to the user factor levels. The inference is made with an auxiliar
    inference mechanism in case the number of actual new instances does not surpass an inferior threshold.

    Args:
        existing_instances: a dict(str, list(int)), where k is the name of a factor or dimension in the model, and v is
            a list of values for it.
        bayesian_net: a BayesianNetwork object used to clasify instances.
        aux_inference_algorithm: an InferenceAlgorithm implementation used when the inferior threshold is not reached.
        inf_threshold: an int that contains the minimal number of instances required to clasify with the BN.
        named_feats: a dict(str, int) where k is a factor name, and v is his corresponding level.
    Returns: a dict(str, int) where k is a dimension name, and v is the infered value for the dimension
    """
    dimensions = copy_all_dims_names()

    if len(existing_instances) >= inf_threshold:  # The threshold is surpassed.
        return bayesian_net.infer_dimensions(dimensions, named_feats)
    else:  # Executes only when the number of new instances is too low to infer with the bayesian network.
        return aux_inference_algorithm.infer_dimensions(existing_instances, named_feats)
