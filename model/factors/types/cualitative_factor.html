from src.model.factors.types.factor import Factor
from src.model.factors.level_calc_types.level_calculation import SelectionCalculation
import numpy as np


class CualitativeFactor(Factor):
    """Factor whose value is calculated using maximun ocurrence, and whose level selection is cualitative based.
    """
    def __init__(self, f_id, f_name, levels):
        """Constructor

        Args:
            f_id: an int that represents the id of the factor.
            f_name: an str that represents the name of the factor.
            levels: a list(int) that contains the levels of the factor.
        """
        self.cualitative_levels = {}
        [self.cualitative_levels.update({l_name: [l_val, 0]}) for l_name, l_val in levels.items()]
        super().__init__(f_id, f_name, self.cualitative_levels, SelectionCalculation())

    def __str__(self):
        return super().__str__() + 'vals={}'.format(self.cualitative_levels)

    def copy(self):
        """Makes a clean copy of the factor.

        Returns:
            a CualitativeFactor.
        """
        levels = {}
        [levels.update({l_name: l_info[0]}) for l_name, l_info in self.cualitative_levels.items()]
        return CualitativeFactor(self.id, self.name, levels)

    def is_valid_level_value(self, value):
        """Checks if the given value is a valid level value.

        Args:
            value: a str, that contains the value to validate.

        Returns:
            a bool with the result of the validation.
        """
        return value in self.cualitative_levels

    def is_valid_event_value(self, value):
        """Checks if the given value is a valid event value.

        Args:
            value: a str, that contains the value to validate.

        Returns:
            a bool with the result of the validation.
        """
        return self.is_valid_level_value(value)

    def format_level_value(self, value):
        """Does not applies any format to the value.

        Args:
            value: a str, that contains the value to format.

        Returns:
            a str with the formated value.
        """
        formated = 'NaN' if (value is np.NaN) else value
        return formated

    def calculate(self):
        """Calculates the level and the value for the factor. The level is the value with most ocurrences.

        Returns:
            a tuple(int, str) if there is at least one ocurrence, or tuple(np.NaN, np.NaN) otherwise. The first value is
                the factor level, and the second one is the level name.
        """
        max_lv_id, max_lv_name, max_ocurr = -1, '', -1
        for lv_name, lv_info in self.cualitative_levels.items():
            if lv_info[1] > max_ocurr:  # lv_info[1] contains the number of ocurrences for the given level.
                max_lv_id, max_lv_name = lv_info[0], lv_name  # lv_info[0] contains the number asociated to the level.
                max_ocurr = lv_info[1]

        if max_ocurr == 0:
            return np.NaN, np.NaN
        return max_lv_id, max_lv_name

    def add_new_ocurrence(self, new_value):
        """Increases the number of ocurrences for the given factor level by one.

        Args:
            new_value: a str that contains the value of the event.

        Returns:
            None
        """
        actual_val = self.cualitative_levels[new_value][1]  # [1] contains the number of ocurrences for the given level.
        self.cualitative_levels[new_value][1] = actual_val + 1
