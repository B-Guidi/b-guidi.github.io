import numpy as np
import datetime

from src.model.factors.types.factor import Factor
from src.model.factors.level_calc_types.level_calculation import RangeCalculation


class DateBasedFactor(Factor):
    def __init__(self, f_id, f_name, levels):
        super().__init__(f_id, f_name, levels, RangeCalculation())
        self.last_date = 0
        self.n_ocurr = 0
        self.time_sum = 0

    def __str__(self):
        return super().__str__() \
               + ', levels={}, last_date={}, time_sum={}'.format(self.levels, self.last_date, self.time_sum)

    def copy(self):
        """
        Makes a clean copy of the factor.

        :return: RangeBasedFactor.
        """
        return DateBasedFactor(self.id, self.name, self.levels)

    def is_valid_level_value(self, value):
        """
        The value is valid if it is greater than 0.

        :param value: int
        :return: bool
        """
        return value > 0

    def is_valid_event_value(self, value):
        """
        Tries to format the value with the specific datetime format. If it raises an exception, then the value is not
            valid.

        :param value: str
        :return: bool
        """
        try:
            datetime.datetime.strptime(value, "%Y-%m-%d %H:%M:%S")
            return True
        except ValueError:
            return False

    def format_level_value(self, value):
        """
        Does not applies any formatting.

        :param value: str
        :return: str
        """
        formated = 'NaN' if (value is np.NaN) else value
        return formated

    def calculate(self):
        """
        Calculates the level and the value for the factor. The responsability of the calculation is delegated to the
            level calculation class.

        :return: tuple(int, int) in case the time_sum and n_ocurr is not zero, or tuple(np.NaN, np.NaN) otherwise.
            The first value is the factor level, and the second one is the factor value.
        """
        if self.time_sum == 0 or self.n_ocurr == 0:
            return np.NaN, np.NaN
        else:
            return self.level_calc.determine_level(self.levels, self.time_sum/self.n_ocurr), \
                   round(self.time_sum/self.n_ocurr)

    def add_new_ocurrence(self, new_value):
        """
        Adds a new ocurrence to the factor. Sums the diference between the last two dates received, stores the last
            date, and increment the number of ocurrences by one.

        :param new_value: str
        :return: None
        """
        new_date = datetime.datetime.strptime(new_value, "%Y-%m-%d %H:%M:%S")
        if self.n_ocurr > 0:
            self.time_sum += round((new_date - self.last_date).total_seconds()/60)  # Convert to minutes
        self.n_ocurr += 1
        self.last_date = new_date
