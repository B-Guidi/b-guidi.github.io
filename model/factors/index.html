"""
Python module that contains classes and functions used to operate with factors.
"""

from src.model.factors.types.cualitative_factor import CualitativeFactor
from src.model.factors.types.date_based_factor import DateBasedFactor
from src.model.factors.types.numeric_factor import NumericFactor

factor_dict = {
    0: NumericFactor(0, 'TiempoLecturaUserStory', [15, 30, 45, 60]),
    1: DateBasedFactor(1, 'TiempoTrabajoUserStory', [30, 60, 120, 180]),
    2: NumericFactor(2, 'ParticipacionesMeetings', [2, 5, 8, 12]),
    3: CualitativeFactor(3, 'Recurso', {'Audio': 0, 'Imagen': 1, 'Texto': 2}),
    4: NumericFactor(4, 'LineasCodigo', [25, 50, 80, 120])
}
"""Existing factors in the model.
"""

name_to_id = {'TiempoLecturaUserStory': 0, 'TiempoTrabajoUserStory': 1, 'ParticipacionesMeetings': 2, 'Recurso': 3,
              'LineasCodigo': 4}
"""Auxiliar dictionary to facilitate factor searching.
"""


def init_factors_matrix():
    """Creates a dictionary where the keys are factors names, and the values are empty list. This method is used when
        querying instances from database.

    Returns:
        a dict(str,list(int))
    """
    factors = {}
    [factors.update({f: []}) for f in name_to_id]
    return factors


def copy_factor_dict():
    """Creates one object of each existing factor, and returns them in a dictionary.

    Returns:
        a dict(int, Factor).
    """
    factors = {}
    [factors.update({f_id: f.copy()}) for f_id, f in factor_dict.items()]
    return factors


def get_total_factors():
    """Returns the number of factors that exist in the model.

    Returns:
        an int
    """
    return len(factor_dict)


def get_factor_name(f_id):
    """Returns the name associated to the factor id.

    Args:
        f_id: an int that contains the factor id.
    Returns:
        a str
    """
    if f_id in factor_dict:
        return factor_dict[f_id].get_name()
    return None


def get_cardinality(f_id):
    """Returns the cardinality of the factor associated to the id.

    Args:
        f_id: an int that contains the factor id.
    Returns:
        an int
    """
    if f_id in factor_dict:
        return factor_dict[f_id].get_cardinality()
    raise KeyError("There is no factor for the id '{}'".format(f_id))


def get_level(f_name, value):
    """Gets the level for a given value in the received factor

    Args:
        f_name: a str that contains the factor name.
        value: value whose level is going to be obtained.
    Returns:
        an int.
    """
    if f_name in name_to_id and is_valid_level_value(f_name, value):
        return factor_dict[name_to_id[f_name]].get_level(value)
    return None


def format_level_value(f_name, value):
    """Formats the given value with the factor implementation.

    Args:
        f_name: a str that contains the factor name.
        value: value formatted.
    Returns:
        the formated value.
    """
    if f_name in name_to_id:
        return factor_dict[name_to_id[f_name]].format_level_value(value)
    return None


def is_valid_level_value(f, value, search_by_name=True):
    """Checks if the given value is a valid level value for the factor.

    Args:
        f: name or id of the factor
        value: value to validate.
        search_by_name: bool that determines how the factor search is made.
    Returns:
        a bool
    """
    # Search by id.
    if not search_by_name:
        if f in factor_dict:  # Checks if there is a factor with the id.
            f = factor_dict[f].get_name()
        else:
            return False

    # Search by name.
    if f in name_to_id:  # Checks if there is a factor with the name.
        return factor_dict[name_to_id[f]].is_valid_level_value(value)

    return False


def is_valid_event_value(f_id, value):
    if f_id in factor_dict:
        return factor_dict[f_id].is_valid_event_value(value)


def are_valid_features(features):
    """Checks if the features values received are valid or not.

    Args:
        features: a dict(str,Any) that contains factors names and values for them.
    :return: bool
    """
    for f_name, f_val in features.items():
        if f_name not in name_to_id or not is_valid_level_value(f_name, f_val):
            return False
    return True


def exist_factor(factor_id):
    return factor_id in factor_dict
