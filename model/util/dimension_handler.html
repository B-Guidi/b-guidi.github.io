dim_info = {
    0: {'name': 'Percepcion',
        'values': {
            -3: 'Intuitivo-Alto', -2: 'Intuitivo-Medio', -1: 'Intuitivo-Bajo', 0: 'Neutro',
            1: 'Sensitivo-Bajo', 2: 'Sensitivo-Medio', 3: 'Sensitivo-Alto'}},
    1: {'name': 'Entendimiento',
        'values': {
            -3: 'Secuencial-Alto', -2: 'Secuencial-Medio', -1: 'Secuencial-Bajo', 0: 'Neutro',
            1: 'Global-Bajo', 2: 'Global-Medio', 3: 'Global-Alto'}},
    2: {'name': 'Procesamiento',
        'values': {
            -3: 'Reflexivo-Alto', -2: 'Reflexivo-Medio', -1: 'Reflexivo-Bajo', 0: 'Neutro',
            1: 'Activo-Bajo', 2: 'Activo-Medio', 3: 'Activo-Alto'}},
    3: {'name': 'Recepcion',
        'values': {0: 'Verbal', 1: 'Visual'}}
}
"""Dictionary that contains the name and the possible levels for each defined dimension.
"""

name_to_id_relation = {'Percepcion': 0, 'Entendimiento': 1, 'Procesamiento': 2, 'Recepcion': 3}


def copy_all_names():
    """Creates an empty dict with (k, v) items where k is the name of a dimension, and v is an empty list.

    Returns:
        a dict(str, list(int))
    """
    r = {}
    [r.update({d_val['name']: []}) for d_val in dim_info.values()]
    return r


def valid_dimensions(dimensions):
    for d_name, d_val in dimensions.items():
        if d_name not in name_to_id_relation or not valid_level(name_to_id_relation[d_name], d_val):
            return False
    return True


def valid_level(dim_id, level):
    return level in dim_info[dim_id]['values']


def get_dimension_name(dim_id):
    return dim_info[dim_id]['name']


def get_dimension_card(dim_id):
    return len(dim_info[dim_id]['values'])


def get_total_dimensions():
    return len(dim_info)
